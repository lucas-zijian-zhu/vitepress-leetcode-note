# [417. Pacific Atlantic Water Flow](https://leetcode.com/problems/pacific-atlantic-water-flow/description/) <Badge type="warning" text="Medium" />

There is an <code>m x n</code> rectangular island that borders both the **Pacific Ocean** and **Atlantic Ocean**. The Pacific Ocean touches the island's left and top edges, and the Atlantic Ocean touches the island's right and bottom edges.

The island is partitioned into a grid of square cells. You are given an <code>m x n</code> integer matrix <code>heights</code> where <code>heights[r][c]</code> represents the **height above sea level** of the cell at coordinate <code>(r, c)</code>.

The island receives a lot of rain, and the rain water can flow to neighboring cells directly north, south, east, and west if the neighboring cell's height is **less than or equal to** the current cell's height. Water can flow from any cell adjacent to an ocean into the ocean.

Return *a **2D list** of grid coordinates <code>result</code> where <code>result[i] = [r<sub>i</sub>, c<sub>i</sub>]</code> denotes that rain water can flow from cell <code>(r<sub>i</sub>, c<sub>i</sub>)</code> to **both** the Pacific and Atlantic oceans*.

**Example 1:**

```
Input: heights = [[1,2,2,3,5],[3,2,3,4,4],[2,4,5,3,1],[6,7,1,4,5],[5,1,1,2,4]]
Output: [[0,4],[1,3],[1,4],[2,2],[3,0],[3,1],[4,0]]
Explanation: The following cells can flow to the Pacific and Atlantic oceans, as shown below:
[0,4]: [0,4] -> Pacific Ocean
       [0,4] -> Atlantic Ocean
...
```

**Example 2:**

```
Input: heights = [[1]]
Output: [[0,0]]
Explanation: The water can flow from the only cell to the Pacific and Atlantic oceans.
```

**Constraints:**

- <code>m == heights.length</code>
- <code>n == heights[i].length</code>
- <code>1 <= m, n <= 200</code>
- <code>0 <= heights[i][j] <= 10<sup>5</sup></code>

## 💡 Approach: DFS from Both Oceans

### Intuition

Instead of checking from each cell whether it can reach both oceans, **start from each ocean boundary** and mark all cells reachable from that ocean (water flows *up* from ocean to higher/equal cells). A cell is in the answer if it is reachable from **both** Pacific and Atlantic.

### Algorithm

1. Create <code>pacific</code> and <code>atlantic</code> boolean arrays to mark reachable cells.
2. **DFS from Pacific boundary**: top row and left column. Mark reachable cells (flow from lower/equal to higher).
3. **DFS from Atlantic boundary**: bottom row and right column.
4. Return all cells that are reachable from both oceans.

### Key Points

- Reverse flow: from ocean (edge) inward, we move to cells with height ≥ current.
- Two separate DFS passes, then collect cells where both boolean arrays are true.
- Similar idea to [130. Surrounded Regions](/graph-general/130) — DFS from boundaries.

```typescript [TypeScript]
function pacificAtlantic(heights: number[][]): number[][] {
    const rows = heights.length;
    const cols = heights[0].length;
    const pacific: boolean[][] = Array.from({ length: rows }, () => Array(cols).fill(false));
    const atlantic: boolean[][] = Array.from({ length: rows }, () => Array(cols).fill(false));
    const dirs = [[1, 0], [0, 1], [-1, 0], [0, -1]];

    function dfs(r: number, c: number, visited: boolean[][]): void {
        if (visited[r][c]) return;
        visited[r][c] = true;
        for (const [dr, dc] of dirs) {
            const nr = r + dr, nc = c + dc;
            if (nr >= 0 && nr < rows && nc >= 0 && nc < cols && heights[nr][nc] >= heights[r][c]) {
                dfs(nr, nc, visited);
            }
        }
    }

    for (let c = 0; c < cols; c++) {
        dfs(0, c, pacific);
        dfs(rows - 1, c, atlantic);
    }
    for (let r = 0; r < rows; r++) {
        dfs(r, 0, pacific);
        dfs(r, cols - 1, atlantic);
    }

    const result: number[][] = [];
    for (let r = 0; r < rows; r++) {
        for (let c = 0; c < cols; c++) {
            if (pacific[r][c] && atlantic[r][c]) {
                result.push([r, c]);
            }
        }
    }
    return result;
}
```

```kotlin [Kotlin]
class Solution {
    fun pacificAtlantic(heights: Array<IntArray>): List<List<Int>> {
        val rows = heights.size
        val cols = heights[0].size
        val pacific = Array(rows) { BooleanArray(cols) }
        val atlantic = Array(rows) { BooleanArray(cols) }
        val dirs = arrayOf(intArrayOf(1, 0), intArrayOf(0, 1), intArrayOf(-1, 0), intArrayOf(0, -1))

        fun dfs(r: Int, c: Int, visited: Array<BooleanArray>) {
            if (visited[r][c]) return
            visited[r][c] = true
            for (d in dirs) {
                val nr = r + d[0]
                val nc = c + d[1]
                if (nr in 0 until rows && nc in 0 until cols && heights[nr][nc] >= heights[r][c]) {
                    dfs(nr, nc, visited)
                }
            }
        }

        for (c in 0 until cols) {
            dfs(0, c, pacific)
            dfs(rows - 1, c, atlantic)
        }
        for (r in 0 until rows) {
            dfs(r, 0, pacific)
            dfs(r, cols - 1, atlantic)
        }

        return (0 until rows).flatMap { r ->
            (0 until cols).mapNotNull { c ->
                if (pacific[r][c] && atlantic[r][c]) listOf(r, c) else null
            }
        }
    }
}
```

```java [Java]
class Solution {
    private int[][] heights;
    private int rows, cols;
    private static final int[][] dirs = {{1, 0}, {0, 1}, {-1, 0}, {0, -1}};

    public List<List<Integer>> pacificAtlantic(int[][] heights) {
        this.heights = heights;
        rows = heights.length;
        cols = heights[0].length;
        boolean[][] pacific = new boolean[rows][cols];
        boolean[][] atlantic = new boolean[rows][cols];

        for (int c = 0; c < cols; c++) {
            dfs(0, c, pacific);
            dfs(rows - 1, c, atlantic);
        }
        for (int r = 0; r < rows; r++) {
            dfs(r, 0, pacific);
            dfs(r, cols - 1, atlantic);
        }

        List<List<Integer>> result = new ArrayList<>();
        for (int r = 0; r < rows; r++) {
            for (int c = 0; c < cols; c++) {
                if (pacific[r][c] && atlantic[r][c]) {
                    result.add(Arrays.asList(r, c));
                }
            }
        }
        return result;
    }

    private void dfs(int r, int c, boolean[][] visited) {
        if (visited[r][c]) return;
        visited[r][c] = true;
        for (int[] d : dirs) {
            int nr = r + d[0], nc = c + d[1];
            if (nr >= 0 && nr < rows && nc >= 0 && nc < cols && heights[nr][nc] >= heights[r][c]) {
                dfs(nr, nc, visited);
            }
        }
    }
}
```

```python [Python]
class Solution:
    def pacificAtlantic(self, heights: List[List[int]]) -> List[List[int]]:
        rows, cols = len(heights), len(heights[0])
        pacific = [[False] * cols for _ in range(rows)]
        atlantic = [[False] * cols for _ in range(rows)]
        dirs = [(1, 0), (0, 1), (-1, 0), (0, -1)]

        def dfs(r: int, c: int, visited: List[List[bool]]) -> None:
            if visited[r][c]:
                return
            visited[r][c] = True
            for dr, dc in dirs:
                nr, nc = r + dr, c + dc
                if 0 <= nr < rows and 0 <= nc < cols and heights[nr][nc] >= heights[r][c]:
                    dfs(nr, nc, visited)

        for c in range(cols):
            dfs(0, c, pacific)
            dfs(rows - 1, c, atlantic)
        for r in range(rows):
            dfs(r, 0, pacific)
            dfs(r, cols - 1, atlantic)

        return [[r, c] for r in range(rows) for c in range(cols) if pacific[r][c] and atlantic[r][c]]
```

## Complexity

- **Time:** O(m × n) — each cell visited at most once per ocean.
- **Space:** O(m × n) — boolean arrays and recursion stack.
