# [23. Merge k Sorted Lists](https://leetcode.com/problems/merge-k-sorted-lists/description/?envType=study-plan-v2&envId=top-interview-150) <Badge type="danger" text="Hard" />

You are given an array of <code>k</code> linked-lists <code>lists</code>, each linked-list is sorted in ascending order.

Merge all the linked-lists into one sorted linked-list and return it.

**Example 1:** 

```
Input: lists = [[1,4,5],[1,3,4],[2,6]]
Output: [1,1,2,3,4,4,5,6]
Explanation: The linked-lists are:
[
  1->4->5,
  1->3->4,
  2->6
]
merging them into one sorted linked list:
1->1->2->3->4->4->5->6
```

**Example 2:** 

```
Input: lists = []
Output: []
```

**Example 3:** 

```
Input: lists = [[]]
Output: []
```

**Constraints:** 

- <code>k == lists.length</code>
- <code>0 <= k <= 10^4</code>
- <code>0 <= lists[i].length <= 500</code>
- <code>-10^4 <= lists[i][j] <= 10^4</code>
- <code>lists[i]</code> is sorted in **ascending order** .
- The sum of <code>lists[i].length</code> will not exceed <code>10^4</code>.

## ðŸ’¡ Approach: Divide and Conquer (Iterative)

### Intuition

We need to merge k sorted linked lists efficiently. The simplest approach would be to merge them one by one, but that's inefficient with O(N Ã— k) time complexity.

**Better approach**: Use **divide and conquer** iteratively to merge pairs of lists, reducing the time complexity to O(N log k).

### Strategy

Think of this as building a **merge tree** iteratively:
- **Level 1**: Merge pairs of original lists â†’ k/2 merged lists
- **Level 2**: Merge pairs of level-1 results â†’ k/4 merged lists  
- **Level 3**: Continue until only 1 final merged list remains

### Algorithm Steps

1. **Initialize**: Filter out null lists
2. **Iterate**: While there's more than one list:
   - Merge pairs of lists (i-th with (i+1)-th)
   - Store merged results in a new array
   - Replace current array with merged results
3. **Return**: The single remaining merged list

This is essentially the same pattern as **merge sort** but implemented iteratively, avoiding recursion stack overhead.

```typescript [TypeScript]
function mergeKLists(lists: Array<ListNode | null>): ListNode | null {
    if (lists.length === 0) return null;
    
    // Iterative approach: merge pairs until only one list remains
    let mergedLists = lists.filter(list => list !== null);
    
    while (mergedLists.length > 1) {
        const nextMerged: Array<ListNode | null> = [];
        
        // Merge pairs of lists
        for (let i = 0; i < mergedLists.length; i += 2) {
            const list1 = mergedLists[i];
            const list2 = (i + 1 < mergedLists.length) ? mergedLists[i + 1] : null;
            nextMerged.push(mergeTwoLists(list1, list2));
        }
        
        mergedLists = nextMerged;
    }
    
    return mergedLists[0] || null;
}

function mergeTwoLists(list1: ListNode | null, list2: ListNode | null): ListNode | null {
    const dummy = new ListNode(0);
    let tail = dummy;
    let l1 = list1;
    let l2 = list2;
    
    // Compare and merge nodes
    while (l1 && l2) {
        if (l1.val <= l2.val) {
            tail.next = l1;
            l1 = l1.next;
        } else {
            tail.next = l2;
            l2 = l2.next;
        }
        tail = tail.next!;
    }
    
    // Attach remaining nodes
    tail.next = l1 || l2;
    return dummy.next;
}
```

```java [Java]
class Solution {
    public ListNode mergeKLists(ListNode[] lists) {
        if (lists.length == 0) return null;
        
        // Iterative approach: merge pairs until only one list remains
        List<ListNode> mergedLists = new ArrayList<>();
        for (ListNode list : lists) {
            if (list != null) mergedLists.add(list);
        }
        
        while (mergedLists.size() > 1) {
            List<ListNode> nextMerged = new ArrayList<>();
            
            // Merge pairs of lists
            for (int i = 0; i < mergedLists.size(); i += 2) {
                ListNode list1 = mergedLists.get(i);
                ListNode list2 = (i + 1 < mergedLists.size()) ? mergedLists.get(i + 1) : null;
                nextMerged.add(mergeTwoLists(list1, list2));
            }
            
            mergedLists = nextMerged;
        }
        
        return mergedLists.isEmpty() ? null : mergedLists.get(0);
    }
    
    // Helper: merge two sorted linked lists
    private ListNode mergeTwoLists(ListNode list1, ListNode list2) {
        ListNode dummy = new ListNode(0);
        ListNode tail = dummy;
        
        // Compare and merge nodes
        while (list1 != null && list2 != null) {
            if (list1.val <= list2.val) {
                tail.next = list1;
                list1 = list1.next;
            } else {
                tail.next = list2;
                list2 = list2.next;
            }
            tail = tail.next;
        }
        
        // Attach remaining nodes
        tail.next = (list1 != null) ? list1 : list2;
        return dummy.next;
    }
}
```

```swift [Swift]
// Definition for singly-linked list.
public class ListNode {
    public var val: Int
    public var next: ListNode?
    public init() { self.val = 0; self.next = nil }
    public init(_ val: Int) { self.val = val; self.next = nil }
    public init(_ val: Int, _ next: ListNode?) { self.val = val; self.next = next }
}

class Solution {
    func mergeKLists(_ lists: [ListNode?]) -> ListNode? {
        guard !lists.isEmpty else { return nil }
        
        // Iterative approach: merge pairs until only one list remains
        var mergedLists = lists.compactMap { $0 }
        
        while mergedLists.count > 1 {
            var nextMerged: [ListNode?] = []
            
            // Merge pairs of lists
            var i = 0
            while i < mergedLists.count {
                let list1 = mergedLists[i]
                let list2 = (i + 1 < mergedLists.count) ? mergedLists[i + 1] : nil
                nextMerged.append(mergeTwoLists(list1, list2))
                i += 2
            }
            
            mergedLists = nextMerged.compactMap { $0 }
        }
        
        return mergedLists.first
    }

    private func mergeTwoLists(_ list1: ListNode?, _ list2: ListNode?) -> ListNode? {
        let dummy = ListNode(0)
        var tail = dummy
        var l1 = list1
        var l2 = list2
        while let a = l1, let b = l2 {
            if a.val <= b.val {
                tail.next = a
                l1 = a.next
            } else {
                tail.next = b
                l2 = b.next
            }
            tail = tail.next!
        }
        tail.next = l1 ?? l2
        return dummy.next
    }
}
```

```python [Python]
class Solution:
    def mergeKLists(self, lists: List[Optional[ListNode]]) -> Optional[ListNode]:
        if not lists:
            return None
        
        # Iterative approach: merge pairs until only one list remains
        merged_lists = [lst for lst in lists if lst is not None]
        
        while len(merged_lists) > 1:
            next_merged = []
            
            # Merge pairs of lists
            for i in range(0, len(merged_lists), 2):
                list1 = merged_lists[i]
                list2 = merged_lists[i + 1] if i + 1 < len(merged_lists) else None
                next_merged.append(self._mergeTwoLists(list1, list2))
            
            merged_lists = next_merged
        
        return merged_lists[0] if merged_lists else None

    def _mergeTwoLists(self, list1: Optional[ListNode], list2: Optional[ListNode]) -> Optional[ListNode]:

                list1 = list1.next
            else:
                tail.next = list2
                list2 = list2.next
```

## Complexity

- **Time Complexity:** O(N log k) where N is the total number of nodes across all lists, and k is the number of linked lists
- **Space Complexity:** O(k) for storing intermediate merged lists (can be optimized to O(1) by merging in-place)
