# [642. Design Search Autocomplete System](https://leetcode.com/problems/design-search-autocomplete-system/)<Badge type="danger" text="Hard" />

Design a search autocomplete system that takes an array of <code>sentences</code> and an array of <code>times</code> (how many times each sentence was typed). Implement:

- **AutocompleteSystem(sentences, times)**  
  - Store each sentence and its frequency.

- **input(c)**  
  - If <code>c == '#'</code>: save the current typed sentence (increment its count) and return an empty list.  
  - Otherwise: append <code>c</code> to the current input, then return the **top 3** historical sentences that have the current string as a **prefix**, ordered by **hot degree** (frequency descending); if tie, by lexicographic order.

**Example:**

```
Input: sentences = ["i love you","island","iroman","i love leetcode"], times = [5,3,2,2]
input("i")  -> ["i love you","island","i love leetcode"]
input(" ")  -> ["i love you","i love leetcode"]
input("a")  -> []
input("#")  -> [], and "i love you" is saved (count becomes 6)
```

**Constraints:**

- At most 100 calls to <code>input</code>.
- <code>1 <= sentences[i].length <= 100</code>, <code>1 <= times[i] <= 50</code>.
- <code>c</code> is a lowercase letter, <code>'#'</code>, or space <code>' '</code>.

## 💡 Approach: Trie + Hot List at Each Node

### Intuition

Use a **Trie** where each node stores a list of **(sentence, count)** that pass through that prefix. When we **insert** or **update** a sentence, we walk the path and at each node add/update the sentence in that node’s list, then sort by (-count, sentence) and keep a bounded size (e.g. top 3 or a bit more). When the user **inputs** a character, move to the child node and return the top 3 from that node’s list.

### Algorithm

1. **TrieNode**: <code>children</code> (map or array for 26 letters + space), and <code>hot</code>: list of <code>(sentence, count)</code>.
2. **Constructor**: Insert each <code>sentences[i]</code> with <code>times[i]</code> into the Trie (at each node, add (sentence, count), sort, trim).
3. **input(c)**:
   - If <code>c == '#'</code>: insert/update current buffer in the Trie, clear buffer, return <code>[]</code>.
   - Else: append <code>c</code> to buffer, move to the child node, return top 3 from that node’s <code>hot</code> list.

### Key Points

- When updating counts for an existing sentence, remove the old (sentence, count) and add (sentence, newCount), then re-sort and trim.
- Handle space: use index 26 or a map keyed by character (a–z and space).

```typescript [TypeScript]
class AutocompleteSystem {
    private root: TrieNode;
    private cur: TrieNode | null;
    private buf: string;
    private count: Map<string, number> = new Map();

    constructor(sentences: string[], times: number[]) {
        this.root = new TrieNode();
        this.cur = this.root;
        this.buf = "";
        for (let i = 0; i < sentences.length; i++) {
            this.count.set(sentences[i], (this.count.get(sentences[i]) ?? 0) + times[i]);
            this.insert(sentences[i]);
        }
    }

    input(c: string): string[] {
        if (c === "#") {
            if (this.buf.length) {
                const s = this.buf;
                this.count.set(s, (this.count.get(s) ?? 0) + 1);
                this.insert(s);
            }
            this.buf = "";
            this.cur = this.root;
            return [];
        }
        this.buf += c;
        const idx = c === " " ? 26 : c.charCodeAt(0) - 97;
        if (!this.cur!.children[idx]) this.cur!.children[idx] = new TrieNode();
        this.cur = this.cur!.children[idx];
        return this.cur!.hot.slice(0, 3).map((x) => x.s);
    }

    private insert(sentence: string): void {
        const cnt = this.count.get(sentence)!;
        let node = this.root;
        for (const c of sentence) {
            const idx = c === " " ? 26 : c.charCodeAt(0) - 97;
            if (!node.children[idx]) node.children[idx] = new TrieNode();
            node = node.children[idx];
            this.updateHot(node, sentence, cnt);
        }
        this.updateHot(node, sentence, cnt);
    }

    private updateHot(node: TrieNode, s: string, cnt: number): void {
        const list = node.hot;
        const i = list.findIndex((x) => x.s === s);
        if (i >= 0) list.splice(i, 1);
        list.push({ s, c: cnt });
        list.sort((a, b) => b.c !== a.c ? b.c - a.c : a.s.localeCompare(b.s));
        if (list.length > 3) list.length = 3;
    }
}
class TrieNode {
    children: (TrieNode | null)[] = Array(27).fill(null);
    hot: { s: string; c: number }[] = [];
}
```

```kotlin [Kotlin]
class AutocompleteSystem(sentences: Array<String>, times: IntArray) {
    private val root = TrieNode()
    private var cur: TrieNode = root
    private var buf = StringBuilder()
    private val count = mutableMapOf<String, Int>()

    init {
        for (i in sentences.indices) {
            count[sentences[i]] = (count[sentences[i]] ?: 0) + times[i]
            insert(sentences[i])
        }
    }

    fun input(c: Char): List<String> {
        if (c == '#') {
            if (buf.isNotEmpty()) {
                val s = buf.toString()
                count[s] = (count[s] ?: 0) + 1
                insert(s)
            }
            buf.clear()
            cur = root
            return emptyList()
        }
        buf.append(c)
        val idx = if (c == ' ') 26 else c - 'a'
        if (cur.children[idx] == null) cur.children[idx] = TrieNode()
        cur = cur.children[idx]!!
        return cur.hot.take(3).map { it.first }
    }

    private fun insert(sentence: String) {
        val cnt = count[sentence]!!
        var node = root
        for (ch in sentence) {
            val idx = if (ch == ' ') 26 else ch - 'a'
            if (node.children[idx] == null) node.children[idx] = TrieNode()
            node = node.children[idx]!!
            updateHot(node, sentence, cnt)
        }
        updateHot(node, sentence, cnt)
    }

    private fun updateHot(node: TrieNode, s: String, cnt: Int) {
        node.hot.removeAll { it.first == s }
        node.hot.add(s to cnt)
        node.hot.sortWith(compareBy({ -it.second }, { it.first }))
        if (node.hot.size > 3) node.hot = node.hot.take(3).toMutableList()
    }
}
class TrieNode {
    val children = arrayOfNulls<TrieNode>(27)
    val hot = mutableListOf<Pair<String, Int>>()
}
```

```java [Java]
class AutocompleteSystem {
    private final TrieNode root = new TrieNode();
    private TrieNode cur;
    private StringBuilder buf = new StringBuilder();
    private Map<String, Integer> count = new HashMap<>();

    public AutocompleteSystem(String[] sentences, int[] times) {
        cur = root;
        for (int i = 0; i < sentences.length; i++) {
            count.merge(sentences[i], times[i], Integer::sum);
            insert(sentences[i]);
        }
    }

    public List<String> input(char c) {
        if (c == '#') {
            if (buf.length() > 0) {
                String s = buf.toString();
                count.merge(s, 1, Integer::sum);
                insert(s);
            }
            buf.setLength(0);
            cur = root;
            return List.of();
        }
        buf.append(c);
        int idx = c == ' ' ? 26 : c - 'a';
        if (cur.children[idx] == null) cur.children[idx] = new TrieNode();
        cur = cur.children[idx];
        return cur.hot.stream().limit(3).map(p -> p.s).toList();
    }

    private void insert(String sentence) {
        int cnt = count.get(sentence);
        TrieNode node = root;
        for (char ch : sentence.toCharArray()) {
            int i = ch == ' ' ? 26 : ch - 'a';
            if (node.children[i] == null) node.children[i] = new TrieNode();
            node = node.children[i];
            updateHot(node, sentence, cnt);
        }
        updateHot(node, sentence, cnt);
    }

    private void updateHot(TrieNode node, String s, int cnt) {
        node.hot.removeIf(p -> p.s.equals(s));
        node.hot.add(new Pair(s, cnt));
        node.hot.sort((a, b) -> a.c != b.c ? Integer.compare(b.c, a.c) : a.s.compareTo(b.s));
        if (node.hot.size() > 3) node.hot = new ArrayList<>(node.hot.subList(0, 3));
    }
}
class TrieNode {
    TrieNode[] children = new TrieNode[27];
    List<Pair> hot = new ArrayList<>();
}
class Pair { String s; int c; Pair(String s, int c) { this.s = s; this.c = c; } }
```

```python [Python]
class TrieNode:
    __slots__ = ('children', 'hot')
    def __init__(self):
        self.children = [None] * 27
        self.hot = []  # list of (sentence, count)

class AutocompleteSystem:
    def __init__(self, sentences: List[str], times: List[int]):
        self.root = TrieNode()
        self.cur = self.root
        self.buf = []
        self.count: Dict[str, int] = {}
        for s, t in zip(sentences, times):
            self.count[s] = self.count.get(s, 0) + t
            self._insert(s)

    def input(self, c: str) -> List[str]:
        if c == '#':
            if self.buf:
                s = ''.join(self.buf)
                self.count[s] = self.count.get(s, 0) + 1
                self._insert(s)
            self.buf = []
            self.cur = self.root
            return []
        self.buf.append(c)
        idx = 26 if c == ' ' else ord(c) - ord('a')
        if self.cur.children[idx] is None:
            self.cur.children[idx] = TrieNode()
        self.cur = self.cur.children[idx]
        return [x[0] for x in self.cur.hot[:3]]

    def _insert(self, sentence: str) -> None:
        cnt = self.count[sentence]
        node = self.root
        for ch in sentence:
            idx = 26 if ch == ' ' else ord(ch) - ord('a')
            if node.children[idx] is None:
                node.children[idx] = TrieNode()
            node = node.children[idx]
            self._update_hot(node, sentence, cnt)
        self._update_hot(node, sentence, cnt)

    def _update_hot(self, node: TrieNode, s: str, cnt: int) -> None:
        node.hot = [(sent, c) for sent, c in node.hot if sent != s]
        node.hot.append((s, cnt))
        node.hot.sort(key=lambda x: (-x[1], x[0]))
        node.hot = node.hot[:3]
```

## Complexity

- **Time:** Insert O(L) per sentence (L = length); input O(1) for traversal + O(1) for top 3.
- **Space:** O(total characters in all sentences).
